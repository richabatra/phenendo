socialButton {shinydashboardPlus}R Documentation

AdminLTE2 social button

Description

Create a social button

Usage

socialButton(url, type = NULL)

Arguments

url

if the button should redirect somewhere.

type

social network name: see here for valid names https://adminlte.io/themes/AdminLTE/pages/UI/buttons.html.

Author(s)

David Granjon, dgranjon@ymail.com

Examples

if (interactive()) {
 library(shiny)
 library(shinydashboard)
 shinyApp(
  ui = dashboardPage(
    dashboardHeader(),
    dashboardSidebar(),
    dashboardBody(
     box(
      title = "Social Buttons",
      status = NULL,
      socialButton(
        url = "http://dropbox.com",
        type = "dropbox"
      ),
      socialButton(
        url = "http://github.com",
        type = "github"
      )
     )
    ),
    title = "Social Buttons"
  ),
  server = function(input, output) { }
 )
}
