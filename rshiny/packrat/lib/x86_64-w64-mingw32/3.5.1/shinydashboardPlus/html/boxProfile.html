boxProfile {shinydashboardPlus}R Documentation

AdminLTE2 box profile

Description

Create box profile

Usage

boxProfile(..., src = NULL, title = NULL, subtitle = NULL)

Arguments

...

any element such as boxProfileItemList.

src

profile image, if any.

title

title.

subtitle

subtitle.

Author(s)

David Granjon, dgranjon@ymail.com

Examples

if (interactive()) {
 library(shiny)
 library(shinydashboard)
 shinyApp(
  ui = dashboardPage(
    dashboardHeader(),
    dashboardSidebar(),
    dashboardBody(
     box(
      title = "Box with profile",
      status = "primary",
      boxProfile(
       src = "https://adminlte.io/themes/AdminLTE/dist/img/user4-128x128.jpg",
       title = "Nina Mcintire",
       subtitle = "Software Engineer",
       boxProfileItemList(
        bordered = TRUE,
        boxProfileItem(
         title = "Followers",
         description = 1322
        ),
        boxProfileItem(
         title = "Following",
         description = 543
        ),
        boxProfileItem(
         title = "Friends",
         description = 13287
        )
       )
      )
     )
    ),
    title = "boxProfile"
  ),
  server = function(input, output) { }
 )
}
