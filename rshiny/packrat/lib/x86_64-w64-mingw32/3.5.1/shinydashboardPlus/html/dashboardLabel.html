dashboardLabel {shinydashboardPlus}R Documentation

AdminLTE2 label

Description

Create a label

Usage

dashboardLabel(..., status = "primary", style = "default")

Arguments

...

any text.

status

label status: "danger", "success", "info", "primary", "warning".

style

label border style: "default" (rounded angles), "circle" or "square".

Author(s)

David Granjon, dgranjon@ymail.com

Examples

if (interactive()) {
 library(shiny)
 library(shinydashboard)
 shinyApp(
  ui = dashboardPage(
    dashboardHeader(),
    dashboardSidebar(),
    dashboardBody(
     dashboardLabel("Label 1", status = "info"),
     dashboardLabel("Label 2", status = "danger", style = "circle"),
     dashboardLabel("Label 3", status = "success", style = "square")
    )
  ),
  server = function(input, output) { }
 )
}
